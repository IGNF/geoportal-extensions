Ajout d'un contrôle de gestionnaire de couches : LayerSwitcher

EXEMPLE D'UTILISATION

                
                    window.onload = function () {
                        var map = new ol.Map({
                            layers: [
                                new ol.layer.Tile({
                                    source : new ol.source.GeoportalWMTS({
                                        layer: "ORTHOIMAGERY.ORTHOPHOTOS"
                                    }),
                                    opacity : 0.8
                                }),
                                new ol.layer.Tile({
                                    source : new ol.source.GeoportalWMTS({
                                        layer: "GEOGRAPHICALGRIDSYSTEMS.MAPS"
                                    })
                                })
                            ],
                            target: 'map',
                            view: new ol.View({
                                center: [288074.8449901076, 6247982.515792289],
                                zoom: 5
                            })
                        });

                        // ajout d'une autre couche
                        var olStamen = new ol.layer.Tile({
                            source : new ol.source.Stamen({
                                layer : "watercolor"
                            }),
                            opacity : 0.5
                        });
                        map.addLayer(olStamen);

                        // ajout du layerSwitcher avec une description pour la couche Stamen
                        layerSwitcher = new ol.control.LayerSwitcher({
                            layers : [{
                                layer : olStamen,
                                config : {
                                    title : "titre initial Stamen",
                                    description : "description initiale Stamen"
                                }
                            }],
                            options : {
                                collapsed: collapsed
                            }
                        });
                        map.addControl(layerSwitcher);

                        // ajout d'une nouvelle configuration pour la couche Stamen
                        var stamenConf = {
                            title : "Images Stamen",
                            description : "Images Stamen",
                            quicklookUrl : "monurl",
                            legends : [
                                {
                                    url : "monurl"
                                }
                            ],
                            metadata : [
                                {
                                    url : "monurl"
                                }
                            ]
                        };
                        layerSwitcher.addLayer(olStamen, stamenConf);
                    }
