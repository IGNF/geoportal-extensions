Ajout d'un widget de dessin

EXEMPLE D'UTILISATION

                
                    window.onload = function () {
                        var map = new ol.Map({
                            layers: [
                                new LayerWMTS({
                                    layer: "GEOGRAPHICALGRIDSYSTEMS.MAPS",
                                })
                            ],
                            target: 'map',
                            view: new ol.View({
                                center: [288074.8449901076, 6247982.515792289],
                                zoom: 5
                            })
                        });
                        map.addControl(new MousePosition({}));
                    }
