OpenLayers3 - A high-performance, feature-packed library for all your mapping needs.

Ajout simplifié de couche WMS Géoportail - en mode bundle avec auto-appel à l'autoconf

avec personnalisation des paramètres WMS

EXEMPLE D'UTILISATION

                
                    window.onload = function () {
                        var gpLayer = new ol.layer.Tile({
                            source : new ol.source.GeoportalWMS({
                                layer : "ORTHOIMAGERY.ORTHOPHOTOS",
                                olParams: {
                                    url: "http://gpp3-wxs.ign.fr/jhyvi0fgmnuxvfv0zjzorvdn/geoportail/r/wms",
                                    params: {
                                        format : "image/png"
                                    }
                                }
                            })
                        });
                        var map = new ol.Map({
                            layers: [gpLayer],
                            target: 'map',
                            view: new ol.View({
                                center: [288074.8449901076, 6247982.515792289],
                                zoom: 10
                            })
                        });
                        map.addControl(
                            new ol.control.GeoportalAttribution({
                                collapsed : false
                            }
                        ));
                    }
