OpenLayers3 - A high-performance, feature-packed library for all your mapping needs.

Ajout simplifié de couche WMS Géoportail en Lambert 93 - en mode bundle avec auto-appel à l'autoconf, et avec affichage des logos.

EXEMPLE D'UTILISATION

                
                    window.onload = function () {
                        var gpLayer = new ol.layer.Tile({
                            source: ol.source.GeoportalWMTS({
                                layer : "ORTHOIMAGERY.ORTHOPHOTOS",
                                olParams : {
                                    params : {
                                        CRS : "EPSG:2154"
                                    }
                                }
                            })
                        });
                        var map = new ol.Map({
                            layers: [gpLayer],
                            target: 'map',
                            view: new ol.View({
                                projection : "EPSG:2154",
                                center: [600000, 6750000],
                                zoom: 10
                            })
                        });
                        map.addControl(
                            new ol.control.GeoportalAttribution({
                                collapsed : false
                            }
                        ));
                    }
