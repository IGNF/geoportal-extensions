OpenLayers3 - A high-performance, feature-packed library for all your mapping needs.

Ajout simplifié de couche WMTS Géoportail en Lambert 93 - en mode bundle avec auto-appel à l'autoconf

EXEMPLE D'UTILISATION

                
                    window.onload = function () {
                        var gpLayer = new ol.layer.Tile({
                            source: Gp.ol.source.GeoportalWMTS({
                                layer: "ORTHOIMAGERY.ORTHOPHOTOS.BDORTHO.L93"
                            })
                        });
                        var map = new ol.Map({
                            layers: [gpLayer],
                            target: 'map',
                            view: new ol.View({
                                projection : "EPSG:2154",
                                center: [288074.8449901076, 6247982.515792289],
                                zoom: 5
                            })
                        });
                        map.addControl(
                            new Gp.ol.control.GeoportalAttribution({
                                collapsed : false
                            }
                        ));
                    }
