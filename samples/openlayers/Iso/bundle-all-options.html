Ajout d'un widget de calcul d'isochrones

EXEMPLE D'UTILISATION

                
                    window.onload = function () {
                        var map = new ol.Map({
                            layers: [
                                new LayerWMTS({
                                    layer: "GEOGRAPHICALGRIDSYSTEMS.MAPS",
                                })
                            ],
                            target: 'map',
                            view: new ol.View({
                                center: [288074.8449901076, 6247982.515792289],
                                zoom: 5
                            })
                        });
                        var iso = new ol.control.IsoChrone({
                            collapsed : false
                        });
                        map.addControl(iso);
                    }
