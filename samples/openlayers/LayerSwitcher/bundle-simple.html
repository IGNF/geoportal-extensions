OpenLayers - A high-performance, feature-packed library for all your mapping needs.

Ajout d'un contrôle de gestionnaire de couches : LayerSwitcher

EXEMPLE D'UTILISATION

                
                    window.onload = function () {
                        var gpLayer = new ol.layer.Tile({
                            source: Gp.ol.source.GeoportalWMTS({
                                layer: "ORTHOIMAGERY.ORTHOPHOTOS"
                            })
                        });
                        var map = new ol.Map({
                            layers: [gpLayer],
                            target: 'map',
                            view: new ol.View({
                                center: [288074.8449901076, 6247982.515792289],
                                zoom: 5
                            })
                        });
                        var layerSwitcher = new ol.control.LayerSwitcher({});
                        map.addControl(layerSwitcher);
                    }
