Ajout simplifié de couche WMTS Géoportail

EXEMPLE D'UTILISATION

                
                    var layers = [
                        new ol.layer.Tile({
                            source: ol.GeoportalSource.WMTS({
                                layer: "ORTHOIMAGERY.ORTHOPHOTOS",
                                apiKey: "0pc8wopu0q520etd77qi619k"
                            })
                        })
                    ];
                    var map = new ol.Map({
                        layers: layers,
                        target: 'map',
                        view: new ol.View({
                            center: [288074.8449901076, 6247982.515792289],
                            zoom: 5
                        })
                    });
